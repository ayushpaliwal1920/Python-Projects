# replace fxn : uses "to_replace= which one wants to replace" and "value = the value should be" parameter.

var.replace(to_replace=19,value=105)

# many value replaced by 1 value :

var.replace(['S001','S002','S003','S004','S005','S006','S007','S008','S009'],99)


# replace using "regex" parameter  : use for replacing ALphabetical data

var.replace("[A-Za-z]","A",regex=True)

var.replace("[A-Z]","A",regex=True)



# Replace using dictionary :

var.replace({"Name":'[A-Z]'},"Ayush",regex=True)


# replace with backword or forward data : using "method" parameter

var.replace(1,method="ffill")

# limit parameter : replace(wantreplace , bywhom, limit  ) to the limit

var.replace(1,"Ayush",limit=3)


# inplace parameter: changes in original data

var.replace("Student","Ayush",inplace=True)
