
#  Drop missing Data : using 'dropna()' fxn

var.dropna()

#    parameters in drop fxn :

#    axis parameter for droping data along a axis like rows or columns. axis =1(along row),0 for along column


var.dropna(axis=1)

#   remove row which is null for every column : use "how= 'any' or 'all' parameter

var.dropna(how='all') # use of "any" will remove every none value


#   Removing null value along a particular column : use "subset=[]" Parameter.

var.dropna(subset=["City"])

#   Remove null data and create a neew data  = using "inplace=" parameter

var.dropna(inplace=True)
var

#   Thres parameter : remove column which only contains only one null value "thres = 1" for one null value 1 , for containing two null value use 2

var.dropna(thresh= 1)
